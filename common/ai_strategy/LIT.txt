    allowed = {
        original_tag = LIT
    }
    enable = {
        country_exists = SIC
    }
    abort = {
        has_war_with = SIC
    }
    abort_when_not_enabled = yes
    ai_strategy = {
        type = invade
        id = SIC
        value = 200
    }
}

LIT_war_with_henan = {
	allowed = {
		original_tag = LIT
	}
	enable = {
		has_completed_focus = LIT_expand_experiment_land
		country_exists = HEN
	}
	abort = {
		NOT = { country_exists = HEN }
	}

	ai_strategy = {
		type = antagonize
		id = HEN
		value = 200
	}
}

henan_of_war_preparation = {
	allowed = {
		original_tag = LIT
	}
	enable = {
		focus_progress = {
			focus = LIT_henan_or_war
			progress > 0.01
		}
		NOT = {
			has_completed_focus = LIT_henan_or_war
		}
		country_exists = HEN
	}

	abort = {
		OR = {
			NOT = { country_exists = HEN }

			AND = {
				has_completed_focus = LIT_henan_or_war
				LIT = { has_war_with = HEN }
			}
		}
	}

	ai_strategy = {
		type = conquer
		id = HEN
		value = 100
	}
}
