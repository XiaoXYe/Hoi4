# in state scope
remove_claim_by_ROOT_and_add_core_of_ROOT = {
	if = {
		limit = { is_claimed_by = ROOT }
		remove_claim_by = ROOT
	}
	if = {
		limit = { NOT = { is_core_of = ROOT } }
		add_core_of = ROOT
	}
}

add_core_od_PREV_and_transfer_to_PREV = {
	if = {
		limit = { NOT = { is_core_of = PREV } }
		add_core_of = PREV
	}
	if = {
		limit = { NOT = { is_owned_and_controlled_by = PREV } }
		transfer_state_to = PREV
	}
}
